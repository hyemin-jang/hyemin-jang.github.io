n = 6  # λ…Έλ“μ μ΄ κ°μ
costs = [  # [λ…Έλ“a, λ…Έλ“b, aμ—μ„ bλ΅ μ΄λ™ν•λ” λΉ„μ©]
    [0,1,7],
    [0,4,3],
    [0,5,10],
    [1,2,4],
    [1,3,10],
    [1,4,2],
    [1,5,6],
    [2,3,2],
    [3,4,11],
    [3,5,9]

def solution(costs):
    values = [2**31-1 for _ in range(n)]  # λΉ„μ© λ°°μ—΄ : intμ μµλ€κ°’μΌλ΅ μ΄κΈ° μ„¤μ •

    answer = 0

    start = 0  # 0λ² λ…Έλ“μ—μ„ μ‹μ‘
    visited[start] = True
    values[start] = 0
    
    while False in visited:
        # λ…Έλ“ μ™„μ „νƒμƒ‰μΌλ΅ λΉ„μ©λ°°μ—΄μ κ°’μ„ μµμ†κ°’μΌλ΅ κ°±μ‹

            if visited[i[1]] == False and i[0] == start:   # i[0]μ—μ„ i[1]λ΅ κ°€λ” κ²½μ°
                values[i[1]] = min(values[i[1]], i[2])
            if visited[i[0]] == False and i[1] == start:   # i[1]μ—μ„ i[0]λ΅ κ°€λ” κ²½μ°
                values[i[0]] = min(values[i[0]], i[2])

        # λ°©λ¬Έν•μ§€ μ•μ€ λ…Έλ“ μ¤‘ μµμ† λΉ„μ©μΌλ΅ κ° μ μλ” λ…Έλ“ νƒμƒ‰
        refer = 2**31-1        
        for i in range(n):
            if visited[i] == False and values[i] != 0:
                refer = min(refer, values[i])
        answer += refer
        
        # ν•΄λ‹Ή λ…Έλ“ λ°©λ¬Έ μ²΄ν¬
        for i in range(n):
            if visited[i] == False and values[i] == refer:
                visited[i] = True
                start = i  # ν„μ¬ λ…Έλ“λ¥Ό iλ΅ μ΄λ™ν• ν›„ whileλ¬Έ μ¬μ‹¤ν–‰
                break            

	return answer

import sys

n = 6
costs = [
    [0,1,7],
    [0,4,3],
    [0,5,10],
    [1,2,4],
    [1,3,10],
    [1,4,2],
    [1,5,6],
    [2,3,2],
    [3,4,11],
    [3,5,9]    
]

def solution(costs):
    values = [sys.maxsize for _ in range(n)]  # λΉ„μ© λ°°μ—΄ : intμ μµλ€κ°’μΌλ΅ μ΄κΈ° μ„¤μ •
    visited = [False for _ in range(n)]  # λ°©λ¬Έ λ°°μ—΄ : Falseλ΅ μ΄κΈ° μ„¤μ •

    values[start] = 0

        if checkLoc == -1:

        visited[checkLoc] = True

    return values
